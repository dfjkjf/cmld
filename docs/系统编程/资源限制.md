# 资源限制

```c
#include <unistd.h>

/* Get file-specific configuration about descriptor FD.  */
extern long int fpathconf (int __fd, int __name) __THROW;

/* Get the value of the system variable NAME.  */
extern long int sysconf (int __name) __THROW;

printf("_SC_ARG_MAX:        ", sysconf(_SC_ARG_MAX));
printf("_SC_LOGIN_NAME_MAX: ", sysconf(_SC_LOGIN_NAME_MAX));
printf("_SC_OPEN_MAX:       ", sysconf(_SC_OPEN_MAX));
printf("_SC_NGROUPS_MAX:    ", sysconf(_SC_NGROUPS_MAX));
printf("_SC_PAGESIZE:       ", sysconf(_SC_PAGESIZE));
printf("_SC_RTSIG_MAX:      ", sysconf(_SC_RTSIG_MAX));

printf("_PC_NAME_MAX: ", fpathconf(STDIN_FILENO, _PC_NAME_MAX));
printf("_PC_PATH_MAX: ", fpathconf(STDIN_FILENO, _PC_PATH_MAX));
printf("_PC_PIPE_BUF: ", fpathconf(STDIN_FILENO, _PC_PIPE_BUF));

output:
_SC_ARG_MAX:         2097152
_SC_LOGIN_NAME_MAX:  256
_SC_OPEN_MAX:        4096
_SC_NGROUPS_MAX:     65536
_SC_PAGESIZE:        4096
_SC_RTSIG_MAX:       32

_PC_NAME_MAX:  255
_PC_PATH_MAX:  4096
_PC_PIPE_BUF:  4096
```
